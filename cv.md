function createDreamTeam() {
  let arr = [...arguments].flat();

  let name = arr
    .reduce((acc, el) => {
      if (typeof el == "string") {
        acc.push(el.trim().charAt(0).toUpperCase());
      }
      return acc;
    }, [])
    .sort()
    .join("");
  return name;
}
